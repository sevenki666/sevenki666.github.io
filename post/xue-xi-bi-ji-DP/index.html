P1216 数字三角形

dp例题。
我们可以计算到达每个点的最大权值，这个值等于其左上的点与右上的点的最大值加上这个点的权值。
状态转移方程：dp[x][y]=max(dp[x−1][y],dp[x−1][y−1])+arr[x][y]dp[x][y]=max(dp[x-1][y],dp[x-1][y-1])+arr[x][y]dp[x][y]=max(dp[x−1][y],dp[x−1][y−1])+arr[x][y].
最后要注意的是，遍历三角形的底层，找出最大值，最大值即为答案。
Code

#include<bits/stdc++.h>
#define MAXN 1002
using namespace std;
int arr[MAXN][MAXN];
int dp[MAXN][MAXN];
int n,ans;
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=i;j++)
			scanf("%d",&arr[i][j]);
	dp[1][1]=arr[1][1];//Don't forget it!
	for(int i=2;i<=n;i++)
		for(int j=1;j<=i;j++)
			dp[i][j]=max(dp[i-1][j],dp[i-1][j-1])+arr[i][j];
	for(int i=1;i<=n;i++)
		ans=max(ans,dp[n][i]);
	printf("%d",ans);
	return 0;
}

U145902 黑熊过河

建立数组dpdpdp，表示到达某一点时所剩的最大能量。
数组arrarrarr表示石墩上食物的能量。
由于熊一次可以跳1~2级，所以起跳前的最大能量是max(dp[i−1],dp[i−2])max(dp[i-1],dp[i-2])max(dp[i−1],dp[i−2])。

如果这个值小于起跳所需要的能量，那么熊就无法安全着陆。
（Q：如果不跳到这个点呢？那是不是还有成立的可能？
A：max(dp[i−1],dp[i−2])max(dp[i-1],dp[i-2])max(dp[i−1],dp[i−2])中必定有一个是熊在跳跃时候落下的点。既然它们两个都满足不了，那么跳跃一定会在这里终止。）
直接输出NO，退出程序。
否则，熊能跳到这上面的最大能量就等于max(dp[i−1],dp[i−2])−q+arr[i]max(dp[i-1],dp[i-2])-q+arr[i]max(dp[i−1],dp[i−2])−q+arr[i]。
Code

#include<bits/stdc++.h>
#define MAXN 1000005
using namespace std;
int arr[MAXN],p,q,n,dp[MAXN];
int main(){
	scanf("%d%d%d",&p,&q,&n);
	for(int i=1;i<=n;i++)scanf("%d",&arr[i]);
	dp[0]=p;
	if(p<q){
        printf("NO");
        return 0;
    }
	dp[1]=dp[0]-q+arr[1];
	for(int i=2;i<=n+1;i++){
		if(max(dp[i-1],dp[i-2])<q){printf("NO");return 0;}
		dp[i]=max(dp[i-1]+arr[i]-q,dp[i-2]+arr[i]-q);
	}
	printf("%d",dp[n+1]);
	return 0;
}

U145904 攀登宝塔

本题有两种向上的方式，一种是爬，一种是飞。
相应的，我们也可以分别使用2个数组记录当前状态。
数组arrarrarr表示爬的每层需要时间。
数组walwalwal表示爬到某层需要的最短时间。
数组flflfl表示飞到某层需要的最短时间。
飞的话，那么飞之前那一个格子必定是走来的。飞的那一格不耗时。
由此可得状态转移方程：fl[i]=min(wal[i−1],wal[i−2])fl[i]=min(wal[i-1],wal[i-2])fl[i]=min(wal[i−1],wal[i−2])
相应的，如果走的话，上一格有可能是飞过来的，也有可能是走过来的。
由此可得状态转移方程：wal[i]=min(fl[i−1],wal[i−1])+arr[i]wal[i]=min(fl[i-1],wal[i-1])+arr[i]wal[i]=min(fl[i−1],wal[i−1])+arr[i]。
最后输出时，输出两种状态的结果的最小值。
Code

#include<bits/stdc++.h>
#define MAXN 10066
using namespace std;
int n,arr[MAXN],fl[MAXN],wal[MAXN];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&arr[i]);
	}
	fl[1]=0;
	wal[1]=arr[1];
	for(int i=2;i<=n;i++){
		fl[i]=min(wal[i-1],wal[i-2]);
		wal[i]=min(fl[i-1],wal[i-1])+arr[i];
	}
	printf("%d",min(fl[n],wal[n]));
	return 0;
}

P1115 最大子段和

刚看到这个题目，第一反应是：
求出前缀和数组，左指针向右找到最小值，右指针向左找到最大值，相减。
但是要考虑的情况太多了。有没有更简单的方法？
动态规划。
数组arrarrarr为原数组。
数组dpdpdp为到此位为止的答案。
推敲与前面dp[i−1]dp[i-1]dp[i−1]的关系：
当dp[i−1]+dp[i]dp[i-1]+dp[i]dp[i−1]+dp[i]能大于arr[i]arr[i]arr[i]时，那就把arr[i]arr[i]arr[i]拼到段的后面 。
即dp[i]=dp[i−1]+arr[i]dp[i]=dp[i-1]+arr[i]dp[i]=dp[i−1]+arr[i]。
否则，还不如自己创造一个新的段。
即dp[i]=arr[i]dp[i]=arr[i]dp[i]=arr[i]。
注意：

  • 观察数据范围，理论最值为2×105×104=2×1092\times 10^5 \times 10^4=2\times 10^92×105×104=2×109。可能有爆int风险。保险起见开long long。（虽然此题int也能过，但是赛场上还是要注意的）
  • 答案有可能为负数，所以ans在进行比较前，要先将ans初始化为LONG_LONG_MIN。
    Code
#include <bits/stdc++.h>
#define MAXN 200006
using namespace std;
long long dp[MAXN],arr[MAXN],n,ans=LONG_LONG_MIN;
int main(){
	scanf("%lld",&n);
	for(int i=1;i<=n;i++)scanf("%lld",&arr[i]);
	dp[1]=arr[1];
	for(int i=2;i<=n;i++){
		dp[i]=max(dp[i-1]+arr[i],arr[i]);
		ans=max(ans,dp[i]);
	}
	printf("%lld",ans);
	return 0;
}

P2925 [USACO08DEC] Hay For Sale S

此题是01背包的模版题。
唯一 的小小区别在于：物品的价值=物品重量。
直接套即可。
观察数据范围，二维数组会MLE，需要滚动数组优化。
Code

#include <bits/stdc++.h>
#define MAXC 50007
#define MAXH 5005
using namespace std;
int dp[MAXC],h,c,v[MAXH];
int main(){
	scanf("%d%d",&c,&h);
	for(int i=1;i<=h;i++)scanf("%d",&v[i]);
	for(int i=1;i<=h;i++){
		for(int j=c;j>=1;j--){
			if(j>=v[i]){
				dp[j]=max(dp[j],dp[j-v[i]]+v[i]);
			}
		}
	}
	printf("%d",dp[c]);
	return 0;
}

P1048 [NOIP2005 普及组] 采药

比上一道更加模版的01背包。
观察数据范围，开不开滚动数组都无所谓。
直接上Code

#include<bits/stdc++.h>
#define MAXT 1005
#define MAXM 105
using namespace std;
int arr[MAXM][MAXT];
int t,m;
int wt[MAXM],wm[MAXM];
int main(){
	scanf("%d%d",&t,&m);
	for(int i=1;i<=m;i++){
		scanf("%d%d",&wt[i],&wm[i]);
	}
	for(int i=1;i<=m;i++){
		for(int j=1;j<=t;j++){
			arr[i][j]=arr[i-1][j];
			if(j>=wt[i]){
				arr[i][j]=max(arr[i][j],arr[i-1][j-wt[i]]+wm[i]);
			}
			
		}
	}
	printf("%d",arr[m][t]);
	return 1+1-1+1-1-1;//皮一下
}

P1060 [NOIP2006 普及组] 开心的金明

题目很高大上的样子，但是看到输入格式就知道
这还tm是01背包。
物品重量就是价格。
物品价值就是价格×重要度。
数据范围可以支撑二维数组。
Code

#include <bits/stdc++.h>
#define MAXN 30005 //Moneys
#define MAXM 27 //Items
using namespace std;
int dp[MAXM][MAXN],pri[MAXM],imp[MAXM],n,m;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++){
		scanf("%d %d",&pri[i],&imp[i]);
		imp[i]*=pri[i];
	}
	for(int i=1;i<=m;i++){
		for(int j=1;j<=n;j++){
			dp[i][j]=dp[i-1][j];
			if(j>=pri[i])dp[i][j]=max(dp[i][j],dp[i-1][j-pri[i]]+imp[i]);
		}
	}
	printf("%d",dp[m][n]);
	return 0;
}

P1802 5 倍经验日

我的天啊怎么还是背包问题
01背包。不同点在于，01背包不拿这个物品，价值就是0；此问题就算不拿也有一定价值。
数据范围依然可以开二维。
数据范围需要开long long。
Code

#include <bits/stdc++.h>
#define MAXN 1007
using namespace std;
long long los[MAXN],win[MAXN],us[MAXN],dp[MAXN][MAXN],n,x;
int main(){
	scanf("%lld%lld",&n,&x);
	for(int i=1;i<=n;i++){
		scanf("%lld%lld%lld",&los[i],&win[i],&us[i]);
	}
	for(int i=1;i<=n;i++){
		for(int j=0;j<=x;j++){
			if(j<us[i])dp[i][j]=dp[i-1][j]+los[i];
			else dp[i][j]=max(dp[i-1][j]+los[i],dp[i-1][j-us[i]]+win[i]);
		}
	}
	printf("%lld",5*dp[n][x]);
	return 0;
}

P1095 [NOIP2007 普及组] 守望者的逃离

看看题目，有没有一种似曾相识的感觉？！
我们先分析一下。
可以发现，魔法的速度是肯定要快于走路的。
依然通过两个数组记录。
数组dpdpdp表示当前所走的路程。
数组dprdprdpr表示跑步走的路程。
这里主要的数组是dprdprdpr。
首先记录dp[i]=dp[i−1]dp[i]=dp[i-1]dp[i]=dp[i−1]，dpr[i]=dpr[i−1]+17dpr[i]=dpr[i-1]+17dpr[i]=dpr[i−1]+17。如果当前魔法≤10\leq 10≤10，就直接使用魔法。否则，dpdpdp数组不走，增加4点魔法。然后更新dpr[i]=max(dpr[i],dp[i])dpr[i]=max(dpr[i],dp[i])dpr[i]=max(dpr[i],dp[i])。
如果中途到达了指定距离，那就可以break了。
否则，我们需要输出跑得最远的路程。这里dprdprdpr的作用就体现出来了。为了跑到最远距离，最后几秒如果休息无法在时限前达到10点魔法，那么就只能跑。
Code

#include <bits/stdc++.h>
#define MAXT 300006
using namespace std;
int dp[MAXT],m,s,t,dpr[MAXT];
int main(){
	scanf("%d%d%d",&m,&s,&t);
	int i;
	for(i=1;i<=t;i++){
		dp[i]=dp[i-1];
		dpr[i]=dpr[i-1]+17;
		if(m>=10){
			m-=10;
			dp[i]+=60;
		}else {
			m+=4;
		}
		if(dpr[i]<dp[i])dpr[i]=dp[i];
		if(dpr[i]>=s){
			break;
		}
	}
	if(dpr[i]>=s){
		printf("Yes\n%d",i);
	}else{
		printf("No\n%d",dpr[t]);//不需要对比dp[t]，因为循环中已经体现了最大值。
	}
	return 0;
}
